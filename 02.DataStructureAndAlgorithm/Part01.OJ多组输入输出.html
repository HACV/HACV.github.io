          # 1、用scanf

          scanf的%s

          输入：是识别到' '（空格）或者'\n'（换行）就不要了，然后在扫描进的后面自动加上'\0'

          输出：是识别到'\0'就输出完毕

          #include<bits/stdc++.h>
using namespace std;

static const int maxn=1e5+5;
int solve[maxn];

int main()
{
    scanf("%s",solve);
    //第1组测试123 24343
    //第2组测试23134131
    
    printf("%s",solve);
    return 0;
}

          # 1）将string类型转换为字符数组

          方法1）用string类的成员函数copy 注意，其实，C语言输入输出和C++输入输出，最好不要混合输入输出，我这样写只是为了论证这样可以

          #include<iostream>
#include<string> 
#include<cstring> 
using namespace std;

int main()
{
	
	while(1)
	{
		string str;
		cin>>str;
		char test[10]; 
		memset(test,-1,sizeof(test));
		//上面的memset是我故意，用来让自己注意
		//string转换到字符数组后，末尾的'\0'要手动加，不然就不行
		 
		
		str.copy(test,str.size());
		test[str.size()]='\0';
		 
		printf("%s\n",test);
	} 
	return 0;
}
