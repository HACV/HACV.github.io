              #include<stdio.h>
#include<math.h>
#include<stdlib.h>


struct node
{
	double val[3];
} solve[3];


int main()
{



	//第（3-loop）个选手的成绩
	int loop=3;
	while( loop-- )
	{
		printf("请输入第%d名选手的三次成绩\n", (3-loop) );

		//输入第（tag+1）个成绩
		int tag=0;
		while( tag<3 )
		{
			//如果输入错误则要求强制重新输入
			while( 1 )
			{	
				double num;
				scanf("%lf",&num);
				if( (num-0.0)>0.0001 && (num-10.0)<0.0001 )
				{
					solve[loop].val[tag]=num;
					break;//关键之处：用while(1)和break的组合来弄出那样的效果
				}
				else
				{
					printf("输入错误,请重新输入！\n");
				}
			}//#end while( 1 )

			++tag;
		}//#end while( tag<3 )

			
	}//#end while( loop-- )


	loop=3;
	while( loop-- )
	{
		double num=solve[loop].val[0];
		int i=1;
		for(; i<3; ++i)
		{
			if( num<solve[loop].val[i] )
			{
				num=solve[loop].val[i];
			}
		}
		printf("第%d名选手的最好成绩为:%.2lf\n", (3-loop), num );

	}


	return 0;
}
