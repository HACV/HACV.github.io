          # 二、（多进程）并发服务器

          至少就我们目前的知识，你和我发送一个连接请求过来。我（服务器）就在后台给他分配一个与之对应的子进程。所以，serve.c也就可以继续去监听其他的客户端请求了。 （多进程） 这样，我的服务器能够创建多少个子进程，我就能创建多少个连接。 思想虽然是这样的，但是具体实现，还要注意的 比如，父进程的cfd（通讯的吧，，好像）和lfd（连接）都被子进程继承了， 对于子进程，要关闭lfd 父进程，要关闭cfd

          父进程，专心致志的监听去。 还要，记得要回收僵尸子进程，不然多尴尬。

          但是，父进程，在专心致志的监听，可能没时间回收子进程，咋办？ 方法1）监听交给子进程去做，父进程专心回收， 方法2）用线程，可以直接让他分离。 方法3）子进程，注册信号捕捉函数，死了的时候，就能自动调用内核回收函数，就把自己回收了。 和1）比较，效率比1高，不需要再弄个子进程了。

            signal(SIGCHLD, wait_child);

    }

    if (pid == 0) {
        while (1) {
            n = Read(cfd, buf, sizeof(buf));
            if (n == 0) {           //client close
                close(cfd);
                return 0;
            } else if (n == -1) {
                perror("read error");
                exit(1);

                for (i = 0; i < n; i++)
                    buf[i] = toupper(buf[i]);
                
             //下面这几行，是为了知道，客户端，他写了什么（所以，对于服务器来说，毫无秘密）
                write(cfd, buf, n);

            }
        }
    }

          # （2）实际操作

          思想和进程差不多，只是函数变了

          #include <stdio.h>
#include <string.h>
#include <arpa/inet.h>
#include <pthread.h>
#include <ctype.h>
#include <unistd.h>
#include <fcntl.h>

#include "wrap.h"

#define MAXLINE 8192
#define SERV_PORT 8000

struct s_info {                     //定义一个结构体, 将地址结构跟cfd捆绑
    struct sockaddr_in cliaddr;
    int connfd;
};

void *do_work(void *arg)
{
    int n,i;
    struct s_info *ts = (struct s_info*)arg;
    char buf[MAXLINE];
    char str[INET_ADDRSTRLEN];      //#define INET_ADDRSTRLEN 16  可用"[+d"查看

    while (1) {
        n = Read(ts->connfd, buf, MAXLINE);      //读客户端
        if (n == 0) {
            printf("the client %d closed...\n", ts->connfd);
            break;                       //跳出循环,关闭cfd
        }
        printf("received from %s at PORT %d\n",
                inet_ntop(AF_INET, &(*ts).cliaddr.sin_addr, str, sizeof(str)),
                ntohs((*ts).cliaddr.sin_port));                 //打印客户端信息(IP/PORT)

        for (i = 0; i < n; i++) 
            buf[i] = toupper(buf[i]);                           //小写-->大写

        Write(STDOUT_FILENO, buf, n);                           //写出至屏幕
        Write(ts->connfd, buf, n);                              //回写给客户端
    }
    Close(ts->connfd);

    return (void *)0;
}

int main(void)
{
    struct sockaddr_in servaddr, cliaddr;
    socklen_t cliaddr_len;
    int listenfd, connfd;
    pthread_t tid;
    struct s_info ts[256];      //根据最大线程数创建结构体数组.
    int i = 0;

    listenfd = Socket(AF_INET, SOCK_STREAM, 0);     //创建一个socket, 得到lfd

    bzero(&servaddr, sizeof(servaddr));         //地址结构清零
    servaddr.sin_family = AF_INET;
    servaddr.sin_addr.s_addr = htonl(INADDR_ANY);   //指定本地任意IP
    servaddr.sin_port = htons(SERV_PORT);                           //指定端口号 8000

    Bind(listenfd, (struct sockaddr *)&servaddr, sizeof(servaddr)); //绑定

    Listen(listenfd, 128);      //设置同一时刻链接服务器上限数

    printf("Accepting client connect ...\n");

    while (1) {
        cliaddr_len = sizeof(cliaddr);
        connfd = Accept(listenfd, (struct sockaddr *)&cliaddr, &cliaddr_len);   //阻塞监听客户端链接请求
        ts[i].cliaddr = cliaddr;
        ts[i].connfd = connfd;

        /* 达到线程最大数时，pthread_create出错处理, 增加服务器稳定性 */
        pthread_create(&tid, NULL, do_work, (void*)&ts[i]);
        pthread_detach(tid);
        //子线程分离,防止僵线程产生.防止产生僵尸线程。
        i++;
    }

    return 0;
}

          # 四、总结

          我们上面的多进程和多线程服务器 能不能直接用他来支持高并发啊？？ 比如开个网站

          实际上是不行的！！！因为开销太大。 好一点的处理方式，我的服务器只有一个程序。
          不应该产生这么多个.c
          不需要，你来一个，我就产生一个线程或者进程去处理。 我们后面有更好的处理方式！！！
          就是select活poll或者epoll
          这种才是我们正儿八经去处理，高并发服务器的方法。

          那么是不是多进程或者多线程是不是没用了呢？？ 当你连接的客户端不那么多的场景，你就可以使用多线程和多进程。 比如，智能家电，大概就40-50个 用多进程就好，要是还有省一点，我用多线程 几十，几百还是可以，多进程或者多线程还是可以满足的， 比如，有的Linux线程能起5百多，1千多啥的。
